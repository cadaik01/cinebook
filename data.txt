#rooms
INSERT INTO rooms (theater_id, name, seat_capacity, created_at, updated_at)
SELECT t.id,
       CONCAT('Room ', r.room_no),
       144,
       NOW(), NOW()
FROM theaters t
JOIN (
    SELECT 1 AS room_no UNION ALL
    SELECT 2 UNION ALL
    SELECT 3 UNION ALL
    SELECT 4
) r;

#seats
INSERT INTO seats (
    room_id,
    seat_row,
    seat_number,
    seat_code,
    seat_type,
    created_at,
    updated_at
)
SELECT
    r.id AS room_id,
    rows.seat_row,
    nums.seat_number,
    CONCAT(rows.seat_row, nums.seat_number) AS seat_code,
    CASE
        -- ghế couple: toàn bộ hàng H
        WHEN rows.seat_row = 'H' THEN 'couple'

        -- ghế vip: hàng C-D-E, ghế 7–12
        WHEN rows.seat_row IN ('C','D','E')
             AND nums.seat_number BETWEEN 7 AND 12
        THEN 'vip'

        -- còn lại là ghế thường
        ELSE 'standard'
    END AS seat_type,
    NOW(),
    NOW()
FROM rooms r
CROSS JOIN (
    SELECT 'A' AS seat_row UNION ALL
    SELECT 'B' UNION ALL
    SELECT 'C' UNION ALL
    SELECT 'D' UNION ALL
    SELECT 'E' UNION ALL
    SELECT 'F' UNION ALL
    SELECT 'G' UNION ALL
    SELECT 'H'
) rows
CROSS JOIN (
    SELECT 1 AS seat_number UNION ALL
    SELECT 2 UNION ALL
    SELECT 3 UNION ALL
    SELECT 4 UNION ALL
    SELECT 5 UNION ALL
    SELECT 6 UNION ALL
    SELECT 7 UNION ALL
    SELECT 8 UNION ALL
    SELECT 9 UNION ALL
    SELECT 10 UNION ALL
    SELECT 11 UNION ALL
    SELECT 12 UNION ALL
    SELECT 13 UNION ALL
    SELECT 14 UNION ALL
    SELECT 15 UNION ALL
    SELECT 16 UNION ALL
    SELECT 17 UNION ALL
    SELECT 18
) nums;
